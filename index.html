Investigación 1

La Construcción de una Clase

Una clase es un plano para definir objetos y su estructura interna. Una vez definida, puede reutilizarse sin necesidad de volver a programar todo. Sus ventajas incluyen organización del código, reutilización y facilidad de modificación. Como desventajas, al inicio puede ser difícil entender POO y puede volverse rígida al exigir que todo esté dentro de una clase. 

class Persona {
    void saludar() {
        System.out.println("Hola");
    }
}

public class Main {
    public static void main(String[] args) {
        new Persona().saludar();
    }
}

Miembros de una Clase

Son los elementos que forman parte de una clase: atributos y métodos. Ventajas: organización, reutilización y encapsulación. Desventajas: complejidad para principiantes y mayor consumo de memoria si hay muchos atributos. 

class Persona {
    private String nombre;
    public int edad;
}

Modificadores de Acceso

Determinan qué clases o métodos pueden acceder a los elementos internos de una clase. Ayudan a proteger los datos y organizar la visibilidad. Desventajas: limitar demasiado puede impedir interacción entre clases y cambiar modificadores después puede generar errores. 

class Persona {
    private String n = "Ana";
    String getN() {
        return n;
    }
}

public class Main {
    public static void main(String[] args) {
        System.out.println(new Persona().getN());
    }
}

Static

Un atributo o método estático pertenece a la clase y no a un objeto específico. Todos los objetos comparten el mismo valor. Ventajas: ahorro de memoria, acceso directo y compartición de datos. Desventajas: dependencia global, dificultad para hacer pruebas y limitaciones dentro de métodos estáticos. 

class Ejemplo {
    static int numero = 5;
}

public class Main {
    public static void main(String[] args) {
        System.out.println(Ejemplo.numero);
    }
}

Conversión de Fechas

import javax.swing.JOptionPane;

public class Fechas {
    public static void main(String[] args) {
        String[][] fechas = new String[3][3];
        int totalFechas = 0;

        while (totalFechas < 3) {
            String entrada = JOptionPane.showInputDialog(null,
            "Ingrese una fecha en formato: día, mes, año\nEjemplo: 15, febrero, 1989");

            if (entrada == null) {
                JOptionPane.showMessageDialog(null, "Proceso cancelado");
                System.exit(0);
            }

            entrada = entrada.trim().toLowerCase();
            String[] partes = entrada.split(",");

            if (partes.length != 3) {
                JOptionPane.showMessageDialog(null, "Formato incorrecto");
                continue;
            }

            try {
                int dia = Integer.parseInt(partes[0].trim());
                String mesTexto = partes[1].trim();
                int anio = Integer.parseInt(partes[2].trim());

                int mes = convertirMes(mesTexto);

                if (mes == -1 || dia < 1 || dia > 31 ||
                String.valueOf(anio).length() != 4) {
                    JOptionPane.showMessageDialog(null, "Formato incorrecto");
                    continue;
                }

                fechas[totalFechas][0] = String.valueOf(dia);
                fechas[totalFechas][1] = String.valueOf(mes);
                fechas[totalFechas][2] = String.valueOf(anio);

                JOptionPane.showMessageDialog(null,
                "Fecha registrada: " + dia + " " + mes + " " + anio);

                totalFechas++;

            } catch (NumberFormatException e) {
                JOptionPane.showMessageDialog(null, "Formato incorrecto");
            }
        }

        String fechaMin = fechas[0][0] + "/" + fechas[0][1] + "/" + fechas[0][2];
        String fechaMax = fechaMin;

        for (int i = 1; i < 3; i++) {
            String actual = fechas[i][0] + "/" + fechas[i][1] + "/" + fechas[i][2];

            if (compararFechas(fechas[i], fechas[0]) < 0) fechaMin = actual;
            if (compararFechas(fechas[i], fechas[0]) > 0) fechaMax = actual;
        }

        JOptionPane.showMessageDialog(null,
        "Fecha más baja: " + fechaMin +
        "\nFecha más alta: " + fechaMax);
    }

    public static int convertirMes(String mes) {
        String[] meses = {"enero","febrero","marzo","abril","mayo","junio",
                          "julio","agosto","septiembre","octubre",
                          "noviembre","diciembre"};

        for (int i = 0; i < meses.length; i++) {
            if (mes.equals(meses[i])) return i + 1;
        }
        return -1;
    }

    public static int compararFechas(String[] f1, String[] f2) {
        int dia1 = Integer.parseInt(f1[0]);
        int mes1 = Integer.parseInt(f1[1]);
        int anio1 = Integer.parseInt(f1[2]);

        int dia2 = Integer.parseInt(f2[0]);
        int mes2 = Integer.parseInt(f2[1]);
        int anio2 = Integer.parseInt(f2[2]);

        if (anio1 != anio2) return anio1 - anio2;
        if (mes1 != mes2) return mes1 - mes2;
        return dia1 - dia2;
    }
}

Cálculo de Edad

import javax.swing.JOptionPane;
import java.time.LocalDate;
import java.time.Period;
import java.time.format.DateTimeParseException;

public class Edad {
    public static void main(String[] args) {
        try {
            String nacimientoStr = JOptionPane.showInputDialog(null,
            "Ingrese la fecha de nacimiento en formato: dd/MM/yyyy");

            if (nacimientoStr == null) System.exit(0);

            String actualStr = JOptionPane.showInputDialog(null,
            "Ingrese la fecha actual en formato: dd/MM/yyyy");

            if (actualStr == null) System.exit(0);

            LocalDate nacimiento = parsearFecha(nacimientoStr);
            LocalDate actual = parsearFecha(actualStr);

            if (nacimiento == null || actual == null ||
            actual.isBefore(nacimiento)) {
                JOptionPane.showMessageDialog(null, "Formato incorrecto");
                System.exit(0);
            }

            Period edad = Period.between(nacimiento, actual);

            if (edad.getYears() >= 1) {
                JOptionPane.showMessageDialog(null,
                "La edad es: " + edad.getYears() + " años.");
            } else {
                JOptionPane.showMessageDialog(null,
                "La edad es: " + edad.getMonths() + " meses y " +
                edad.getDays() + " días.");
            }

        } catch (Exception e) {
            JOptionPane.showMessageDialog(null, "Formato incorrecto");
        }
    }

    private static LocalDate parsearFecha(String fecha) {
        try {
            String[] partes = fecha.split("/");

            if (partes.length != 3) return null;

            int dia = Integer.parseInt(partes[0]);
            int mes = Integer.parseInt(partes[1]);
            int anio = Integer.parseInt(partes[2]);

            if (dia < 1 || dia > 31 || mes < 1 || mes > 12 ||
            String.valueOf(anio).length() != 4) {
                return null;
            }

            return LocalDate.of(anio, mes, dia);

        } catch (NumberFormatException | DateTimeParseException e) {
            return null;
        }
    }
}

Año Bisiesto

import javax.swing.JOptionPane;

public class Bisiesto {
    public static void main(String[] args) {
        try {
            String entrada = JOptionPane.showInputDialog(null,
            "Ingrese un año de 4 dígitos");

            if (entrada == null) System.exit(0);

            if (!entrada.matches("\\d{4}")) {
                JOptionPane.showMessageDialog(null, "Formato incorrecto");
                System.exit(0);
            }

            int anio = Integer.parseInt(entrada);
            boolean esBisiesto = false;

            if (anio % 4 == 0) {
                if (anio % 100 == 0) {
                    if (anio % 400 == 0) esBisiesto = true;
                } else {
                    esBisiesto = true;
                }
            }

            if (esBisiesto) {
                JOptionPane.showMessageDialog(null, anio + " es bisiesto.");
            } else {
                JOptionPane.showMessageDialog(null, anio + " no es bisiesto.");
            }

        } catch (Exception e) {
            JOptionPane.showMessageDialog(null, "Formato incorrecto");
        }
    }
}

Salario Semanal

import javax.swing.JOptionPane;

public class Salario {
    public static void main(String[] args) {
        try {
            String horasStr = JOptionPane.showInputDialog(null,
            "Ingrese horas trabajadas:");

            if (horasStr == null) return;

            if (horasStr.length() > 2) return;

            double horas = Double.parseDouble(horasStr);
            if (horas < 0) return;

            String tasaStr = JOptionPane.showInputDialog(null,
            "Ingrese la tasa por hora:");

            if (tasaStr == null) return;

            double tasa = Double.parseDouble(tasaStr);
            if (tasa < 0) return;

            double salarioBruto;
            if (horas <= 40) {
                salarioBruto = horas * tasa;
            } else {
                double extras = horas - 40;
                salarioBruto = (40 * tasa) + (extras * tasa * 1.5);
            }

            double impuesto = (salarioBruto > 750) ? salarioBruto * 0.10 : 0;
            double salarioNeto = salarioBruto - impuesto;

            JOptionPane.showMessageDialog(null,
            "Salario Bruto: " + salarioBruto +
            "\nImpuesto: " + impuesto +
            "\nSalario Neto: " + salarioNeto);

        } catch (Exception e) {
        }
    }
}

import java.sql.Connection;

import java.sql.ResultSet;

public class UsuarioDAO {
    public boolean validar(String usuario, String clave) {
        try (Connection con = DatabaseConnection.conectar()) {
            PreparedStatement stmt =
            con.prepareStatement("SELECT * FROM usuarios WHERE usuario=? AND clave=?");

            stmt.setString(1, usuario);
            stmt.setString(2, clave);

            ResultSet rs = stmt.executeQuery();
            return rs.next();

        } catch (Exception e) {
            return false;
        }
    }

        knowledge.put("mysql", "MySQL es un motor de base de datos.");

    public String responder(String pregunta) {

        for (String key : knowledge.keySet()) {

        }
        return "No entendí la consulta.";

ChatBotPanel.java

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class ChatBotPanel extends JPanel {
    private ChatBotService bot = new ChatBotService();

    public ChatBotPanel() {
        setLayout(new BorderLayout());

        JTextArea area = new JTextArea();
        JTextField campo = new JTextField();

        add(new JScrollPane(area), BorderLayout.CENTER);
        add(campo, BorderLayout.SOUTH);

        campo.addActionListener(new ActionListener() {
            public void actionPerformed(ActionEvent e) {
                String p = campo.getText();
                area.append("Tú: " + p + "\n");
                area.append("Bot: " + bot.responder(p) + "\n\n");
                campo.setText("");
            }
        });
    }
}
